#include <bits/stdc++.h>
using namespace std;

int main() {
	int arraySize;
	cout << "Enter Array Size: ";
	cin >> arraySize;

	int numbers[arraySize];

	for(int i = 0; i < arraySize; i++) 
	{
		cout << "numbers[" << i+1 << "] = ";
		cin >> numbers[i];
	}

	int position;
	cout << "Enter Deleted Position: ";
	cin >> position;

	for(int j = position - 1; j < arraySize - 1; j++) 
	{
		numbers[j] = numbers[j+1];
	}

	arraySize--;

	for(int i = 0; i < arraySize; i++) 
	{
		cout << numbers[i] << " ";
	}

	cout << endl;
	return 0;
}
