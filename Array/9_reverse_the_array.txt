#include <bits/stdc++.h>
using namespace std;

int main() {
	int arraySize;
	
	cout << "Enter Array Size: ";
	cin >> arraySize;

	int numbers[arraySize];

	for(int i = 0; i < arraySize; i++) 
	{
		cout << "numbers[" << i << "] = ";
		cin >> numbers[i];
	}

	for(int j = 0; j < arraySize/2; j++)
	{
		swap(numbers[j], numbers[arraySize-j-1]);
	}

	cout << "After Reverse: " << endl;
	
	for(int k = 0; k < arraySize; k++)
	{
		cout << numbers[k] << " ";
	}

	cout << endl;
	return 0;
}
