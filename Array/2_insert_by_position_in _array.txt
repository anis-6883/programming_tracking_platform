#include <bits/stdc++.h>
using namespace std;

int main() {
	int arraySize;
	cout << "Enter Array Size: ";
	cin >> arraySize;

	int numbers[arraySize * 2];

	for(int i = 0; i < arraySize; i++) {
		cout << "numbers[" << i+1 << "] = ";
		cin >> numbers[i];
	}

	int position, insertNumber;
	cout << "Enter Inserted Position: ";
	cin >> position;
	cout << "Enter Inserted Number: ";
	cin >> insertNumber;

	arraySize++;

	for(int j = arraySize - 1; j >= position; j--) {
		numbers[j] = numbers[j-1];
	}

	numbers[position - 1] = insertNumber;

	for(int i = 0; i < arraySize; i++){
		cout << numbers[i] << " ";
	}

	cout << endl;
	return 0;
}
