#include <iostream>
using namespace std;

int main()
{
    int n;
    cout << "Enter Array Size: ";
    cin >> n;

    int arr[n];

    for(int i=0; i<n; i++)
    {
        cout << "arr[" << i + 1 << "] = ";
        cin >> arr[i];
    }

    cout << "\nArray Elements:" << endl;

    for(int i=0; i<n; i++)
    {
        cout << arr[i] << " ";
    }

    cout << endl;
    return 0;
}
